<script>
const CIGS_PER_PACK = 20;

let startRaw = localStorage.getItem("start");
let price = parseFloat(localStorage.getItem("price")) || 165;
let cigsPerDay = parseFloat(localStorage.getItem("cigs")) || 20;

let start = null;
const startDisplay = document.getElementById("startDisplay");

// ✅ НАДЕЖНЫЙ ПАРСИНГ ДАТЫ
function parseDate(str){
  if(!str) return null;

  // фикс iOS
  const fixed = str.replace("T"," ");
  const d = new Date(fixed);

  return isNaN(d) ? null : d;
}

if (startRaw) {
  start = parseDate(startRaw);

  if(start){
    startDisplay.innerText = formatDate(start);
  } else {
    startDisplay.innerText = "Ошибка даты";
  }
}

function save(){
  const d = document.getElementById("dateInput").value;
  const p = document.getElementById("priceInput").value;
  const c = document.getElementById("cigsInput").value;

  if(d) localStorage.setItem("start", d);
  if(p) localStorage.setItem("price", p);
  if(c) localStorage.setItem("cigs", c);

  location.reload();
}

function update(){
  if (!start) return;

  const now = new Date();

  // ❗ защита от будущей даты
  if(start > now){
    document.getElementById("time").innerText = "0д 0ч 0м";
    return;
  }

  const sec = Math.floor((now - start)/1000);

  const d = Math.floor(sec/86400);
  const h = Math.floor(sec/3600)%24;
  const m = Math.floor(sec/60)%60;

  document.getElementById("time").innerText = `${d}д ${h}ч ${m}м`;

  const saved = (sec * cigsPerDay * price / (CIGS_PER_PACK * 86400));
  document.getElementById("money").innerText = saved.toFixed(2) + " грн";

  const cigs = Math.floor((sec/86400)*cigsPerDay);
  document.getElementById("cigs").innerText = cigs;
}

function formatDate(d){
  return d.toLocaleString("ru-RU", {
    day: "2-digit",
    month: "2-digit",
    year: "numeric",
    hour: "2-digit",
    minute: "2-digit"
  });
}

setInterval(update,1000);
update();
